160 bool IsEvtSelected(aod::CollisionEvSelCent const& collision)
161 {
162  if (collision.alias()[kINT7]) {
163  if (collision.sel7()) {
164  if (zvtxlow < collision.posZ() and collision.posZ() < zvtxup) {
165  return true;
166  }
167  return false;
168  }
169  }
170  return false;
171 }

173 bool matchTrackType(aod::TrackData const& track)
174 {
175  switch (tracktype) {
176  case 1:
177  if (track.isGlobalTrack() != 0)
178  return true;
179  else
180  return false;
181  break;
182  case 2:
183  if (track.isGlobalTrackSDD() != 0)
184  return true;
185  else
186  return false;
187  break;
188  default:
189  return false;
190  }
191 }

193 inline std::tuple<uint8_t, uint8_t> AcceptTrack(aod::TrackData const& track)
194 {
195 
196  uint8_t asone = DPTDPT_FALSE;
197  uint8_t astwo = DPTDPT_FALSE;
198 
199  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
200  if (matchTrackType(track)) {
201  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
202  if (((track.charge() > 0) and (trackonecharge > 0)) or ((track.charge() < 0) and (trackonecharge < 0))) {
203  asone = DPTDPT_TRUE;
204  } else if (((track.charge() > 0) and (tracktwocharge > 0)) or ((track.charge() < 0) and (tracktwocharge < 0))) {
205  astwo = DPTDPT_TRUE;
206  }
207  }
208  }
209  return std::make_tuple(asone, astwo);
210 }

221 inline int GetEtaPhiIndex(float eta, float phi)
222 {
223  int etaix = int((eta - etalow) / etabinwidth);
224  int phiix = int((phi - philow) / phibinwidth);
225  return etaix * phibins + phiix;

332  void process(aod::CollisionEvSelCent const& collision, soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection> const& ftracks)
333  {
334  // LOGF(INFO,"New collision with %d filtered tracks", ftracks.size());
335  fhCentMultB->Fill(collision.centV0M());
336  fhVertexZB->Fill(collision.posZ());
337  int acceptedevent = DPTDPT_FALSE;
338  if (IsEvtSelected(collision)) {
339  acceptedevent = DPTDPT_TRUE;
340  fhCentMultA->Fill(collision.centV0M());
341  fhVertexZA->Fill(collision.posZ());
342  // LOGF(INFO,"New accepted collision with %d filtered tracks", ftracks.size());
343 
344  for (auto& track : ftracks) {
345  /* before track selection */
346  fhPtB->Fill(track.pt());
347  fhEtaB->Fill(track.eta());
348  fhPhiB->Fill(track.phi());
349  fhEtaVsPhiB->Fill(track.phi(), track.eta());
350  fhPtVsEtaB->Fill(track.eta(), track.pt());
351  if (track.charge() > 0) {
352  fhPtPosB->Fill(track.pt());
353  } else {
354  fhPtNegB->Fill(track.pt());
355  }
356 
357  /* track selection */
358  /* tricky because the boolean columns issue */
359  auto [asone, astwo] = AcceptTrack(track);
360  if ((asone == DPTDPT_TRUE) or (astwo == DPTDPT_TRUE)) {
361  /* the track has been accepted */
362  fhPtA->Fill(track.pt());
363  fhEtaA->Fill(track.eta());
364  fhPhiA->Fill(track.phi());
365  fhEtaVsPhiA->Fill(track.phi(), track.eta());
366  fhPtVsEtaA->Fill(track.eta(), track.pt());
367  if (track.charge() > 0) {
368  fhPtPosA->Fill(track.pt());
369  } else {
370  fhPtNegA->Fill(track.pt());
371  }
372  }
373  scannedtracks(asone, astwo);
374  }
375  } else {
376  for (auto& track : ftracks) {
377  scannedtracks(DPTDPT_FALSE, DPTDPT_FALSE);
378  }
379  }
380  acceptedevents(acceptedevent);
381  }
382 };
